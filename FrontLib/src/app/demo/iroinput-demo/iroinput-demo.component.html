Input

It provides the same functionality as a native <input /> enhanced with Material Design styling and animations.

Attributes Details :

  • isRequired : boolean

    To make input field required, enable this flag

  • inputType : string

    Set type of the input

  • placeholder : string

    It is a placeholder string for the input field.

  • value : string

    It is a value given to input field.

  • isReadOnly : boolean

    It makes input field readonly.

  • customFormControl : FormControl

    Override pre-defined validations of input

  • customFormControlData : object

    Input control messages are defined inside this object. For example: if field is required the meesage shown will be given against required key.

  export class IroinputDemoComponent implements OnInit {
  
    // --- iro inputbox
    input_1_placeholder = 'Enter Password';
    input_2_placeholder = 'First Name';
    input_3_placeholder = 'Email Id';
    input_1_required = false;
    input_2_required = true;
    input_1_type = 'password';
    input_2_type = 'text';
    input_3_type = 'email';
    input_1_modal = '';
    input_2_modal = 'Roger';
    input_3_modal = '';
    input_1_readOnly = false;
    input_2_readOnly = true;
  
    customFormControl = new FormControl('', [
      Validators.required,
      Validators.pattern(EMAIL_REGEX)
    ]);
  
    customFormControlData = {
      pattern: 'Please enter a valid email address',
      required: 'Email is Required'
    };
  
    constructor() { }
  
    ngOnInit() {
    }
  
  
  
  }
